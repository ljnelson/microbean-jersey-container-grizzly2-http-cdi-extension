050/**
051 * An {@link AbstractBlockingExtension} that searches for any {@link
052 * HttpServer} instances in the current {@linkplain BeanManager CDI
053 * container} and, for each one, {@linkplain HttpServer#start() starts
054 * it}.
055 *
056 * <p>The CDI container is {@linkplain AbstractBlockingExtension
057 * blocked politely} to keep the container up and running while the
058 * {@link HttpServer}s are running.</p>
059 *
060 * @author <a href="https://about.me/lairdnelson"
061 * target="_parent">Laird Nelson</a>
062 *
063 * @see HttpServer
064 *
065 * @see HttpServer#start()
066 *
067 * @see AbstractBlockingExtension
068 */
069public class HttpServerStartingExtension extends AbstractBlockingExtension {
070
071
072  /*
073   * Instance fields.
074   */
075
076  
077  /**
078   * A {@link Collection} of {@link HttpServer}s that have been
079   * {@linkplain HttpServer#start() started}.
080   *
081   * <p>This field is never {@code null}.</p>
082   *
083   * <h2>Thread Safety</h2>
084   *
085   * <p>This field is not safe for concurrent use by multiple
086   * threads.</p>
087   *
088   * @see #startHttpServers(Object, BeanManager)
089   */
090  private final Collection<HttpServer> startedHttpServers;
091
092
093  /*
094   * Constructors.
095   */
096
097
098  /**
099   * Creates a new {@link HttpServerStartingExtension}.
100   *
101   * @see AbstractBlockingExtension#AbstractBlockingExtension()
102   */
103  public HttpServerStartingExtension() {
104    super();
105    this.startedHttpServers = new LinkedList<>();
106  }
107
108
109  /*
110   * Instance methods.
111   */
112
113
114  /**
115   * Upon startup ({@linkplain Initialized initialization} of the
116   * {@linkplain ApplicationScoped application scope}), looks for
117   * {@link HttpServer} instances in the {@linkplain BeanManager CDI
118   * container} represented by the supplied {@link BeanManager} and
119   * {@linkplain HttpServer#start() starts them}.
120   *
121   * <p>If any {@link HttpServer}s were {@linkplain HttpServer#start()
122   * started}, then the CDI container is {@linkplain
123   * AbstractBlockingExtension politely blocked} until they are
124   * {@linkplain HttpServer#shutdown() stopped}.</p>
125   *
126   * @param event the event signaling that the CDI container has
127   * started; ignored; may be {@code null}
128   *
129   * @param beanManager a {@link BeanManager} supplied by the current
130   * CDI container; may be {@code null} in which case no action will
131   * be taken
132   *
133   * @exception IOException if an error occurs while {@linkplain
134   * HttpServer#start() starting the <code>HttpServer</code>
135   * instances}
136   *
137   * @see HttpServer#start()
138   *
139   * @see AbstractBlockingExtension
140   */
141  private final void startHttpServers(@Observes
142                                      @Initialized(ApplicationScoped.class)
143                                      @Priority(LIBRARY_AFTER - 1)
144                                      final Object event,
145                                      final BeanManager beanManager)
146    throws IOException {
147    if (this.logger.isTraceEnabled()) {
148      this.logger.trace("ENTRY {} {} {}, {}", this.getClass().getName(), "startHttpServers", event, beanManager);
149    }
150    if (beanManager != null) {
151      final Instance<Object> beans = beanManager.createInstance();
152      assert beans != null;
153      final Instance<HttpServer> httpServers = beans.select(HttpServer.class);
154      assert httpServers != null;
155      if (!httpServers.isUnsatisfied()) {
156        synchronized (this.startedHttpServers) {
157          try {
158            for (final HttpServer httpServer : httpServers) {
159              if (httpServer != null) {
160
161                // This (asynchronous) method starts a daemon thread
162                // in the background; see
163                // https://github.com/GrizzlyNIO/grizzly-mirror/blob/2.3.x/modules/http-server/src/main/java/org/glassfish/grizzly/http/server/HttpServer.java#L816
164                // and work backwards to the start() method.  Among
165                // other things, that won't prevent the JVM from
166                // exiting normally.  Think about that for a while.
167                // We'll need another mechanism to block the CDI
168                // container from simply shutting down.
169                if (this.logger.isInfoEnabled()) {
170                  this.logger.info("Starting HttpServer: {}", httpServer);
171                }
172                httpServer.start();
173                
174                // We store our own list of started HttpServers to use
175                // in other methods in this class rather than relying
176                // on Instance<HttpServer> because we don't know what
177                // scope the HttpServer instances in question are.
178                // For example, they might be in Dependent scope,
179                // which would mean every Instance#get() invocation
180                // might create a new one.
181                this.startedHttpServers.add(httpServer);
182                
183              }
184            }
185          } catch (final IOException startMethodFailed) {
186            try {
187              this.stopHttpServers(startMethodFailed, false /* forceRemove */);
188            } catch (final ExecutionException | InterruptedException suppressMe) {
189              if (suppressMe instanceof InterruptedException) {
190                Thread.currentThread().interrupt();
191              }
192              startMethodFailed.addSuppressed(suppressMe);
193            }
194            throw startMethodFailed;
195          }
196          
197          if (this.startedHttpServers.isEmpty()) {
198            this.unblock();
199          }
200          
201        }
202      }
203    }
204    if (logger.isTraceEnabled()) {
205      logger.trace("EXIT {} {}", this.getClass().getName(), "startHttpServers");
206    }
207  }
208
209  /**
210   * Calls {@link HttpServer#shutdown()} on any {@link HttpServer}
211   * instances that were {@linkplain #startHttpServers(Object,
212   * BeanManager) started}.
213   *
214   * @param event the event signaling that the current CDI container
215   * is about to shut down; may be {@code null}; ignored
216   *
217   * @exception ExecutionException if there was a problem {@linkplain
218   * HttpServer#shutdown() shutting down} an {@link HttpServer}
219   *
220   * @exception InterruptedException if the {@link Thread} performing
221   * the shutdown was interrupted
222   *
223   * @see #startHttpServers(Object, BeanManager)
224   *
225   * @see HttpServer#shutdownNow()
226   */
227  private final void stopHttpServers(@Observes
228                                     @BeforeDestroyed(ApplicationScoped.class)
229                                     final Object event)
230    throws ExecutionException, InterruptedException {
231    if (logger.isTraceEnabled()) {
232      logger.trace("ENTRY {} {} {}", this.getClass().getName(), "stopHttpServers", event);
233    }
234    try {
235      this.stopHttpServers(null /* no IOException */, event != null /* forceRemove */);
236    } catch (final IOException willNeverHappen) {
237      throw new AssertionError("Unexpected theoretically impossible java.io.IOException", willNeverHappen);
238    }
239    if (logger.isTraceEnabled()) {
240      logger.trace("EXIT {} {}", this.getClass().getName(), "stopHttpServers");
241    }
242  }
243
244  /**
245   * Calls {@link HttpServer#shutdown()} on any {@link HttpServer}
246   * instances that were {@linkplain #startHttpServers(Object,
247   * BeanManager) started}.
248   *
249   * @param startMethodFailed an {@link IOException} that may have
250   * caused this method to be invoked; if non-{@code null} then any
251   * exceptions encountered during the shutdown process will be
252   * {@linkplain Throwable#addSuppressed(Throwable) added to it}
253   * before it is rethrown; may be {@code null}
254   *
255   * @param forceRemove whether to forcibly remove the {@link HttpServer}
256   * from an internal list of started {@link HttpServer}s regardless of
257   * whether an attempt to shut it down succeeded
258   *
259   * @exception ExecutionException if {@code startMethodFailed} is
260   * {@code null} and the {@link HttpServer#shutdown()} method
261   * resulted in a {@link Future} whose {@link Future#get()} method
262   * threw an {@link ExecutionException}
263   *
264   * @exception InterruptedException if {@code startMethodFailed} is
265   * {@code null} and the {@link Thread} performing the shutdown was
266   * interrupted
267   *
268   * @exception IOException if {@code startMethodFailed} is non-{@code
269   * null}
270   *
271   * @see #startHttpServers(Object, BeanManager)
272   *
273   * @see HttpServer#shutdownNow()
274   */
275  private final void stopHttpServers(final IOException startMethodFailed,
276                                     final boolean forceRemove)
277    throws ExecutionException, InterruptedException, IOException {
278    if (this.logger.isTraceEnabled()) {
279      this.logger.trace("ENTRY {} {} {}, {}", this.getClass().getName(), "stopHttpServers", startMethodFailed, forceRemove);
280    }
281    synchronized (this.startedHttpServers) {
282      if (!this.startedHttpServers.isEmpty()) {
283        final Iterator<HttpServer> iterator = this.startedHttpServers.iterator();
284        assert iterator != null;
285        while (iterator.hasNext()) {
286          final HttpServer httpServer = iterator.next();
287          if (forceRemove) {
288            iterator.remove();
289          }
290          if (httpServer != null && httpServer.isStarted()) {
291            try {
292              final Future<HttpServer> shutdownFuture = httpServer.shutdown();
293              assert shutdownFuture != null;
294              final HttpServer stoppedServer = shutdownFuture.get();
295              if (!forceRemove) {
296                iterator.remove();
297              }
298              assert stoppedServer == httpServer;
299              assert !httpServer.isStarted();
300            } catch (final ExecutionException | InterruptedException | RuntimeException problem) {
301              if (problem instanceof InterruptedException) {
302                Thread.currentThread().interrupt();
303              }
304              if (startMethodFailed != null) {
305                startMethodFailed.addSuppressed(problem);
306              } else {
307                throw problem;
308              }
309            }
310          }
311        }
312      }
313      if (startMethodFailed == null) {
314        if (logger.isDebugEnabled()) {
315          logger.debug("Stopped HttpServer instances because of a normal shutdown");
316        }
317      } else {
318        if (logger.isErrorEnabled()) {
319          logger.error("Stopped HttpServer instances because of a java.io.IOException", startMethodFailed);
320        }
321        throw startMethodFailed;
322      }
323    }
324    if (logger.isTraceEnabled()) {
325      logger.trace("EXIT {} {}", this.getClass().getName(), "stopHttpServers");
326    }
327  }

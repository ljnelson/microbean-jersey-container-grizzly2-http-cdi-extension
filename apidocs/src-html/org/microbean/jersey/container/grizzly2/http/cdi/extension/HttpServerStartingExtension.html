049/**
050 * An {@link AbstractBlockingExtension} that searches for any {@link
051 * HttpServer} instances in the current {@linkplain BeanManager CDI
052 * container} and, for each one, {@linkplain HttpServer#start() starts
053 * it}.
054 *
055 * <p>A {@linkplain #fireBlockingEvent(BeanManager) blocking event is
056 * then fired} to keep the container up and running while the {@link
057 * HttpServer}s are running.</p>
058 *
059 * @author <a href="https://about.me/lairdnelson"
060 * target="_parent">Laird Nelson</a>
061 *
062 * @see HttpServer
063 *
064 * @see HttpServer#start()
065 *
066 * @see AbstractBlockingExtension
067 */
068public class HttpServerStartingExtension extends AbstractBlockingExtension {
069
070
071  /*
072   * Instance fields.
073   */
074
075
076  /**
077   * A {@link Collection} of {@link HttpServer}s that have been
078   * {@linkplain HttpServer#start() started}.
079   *
080   * <p>This field is never {@code null}.</p>
081   *
082   * <h2>Thread Safety</h2>
083   *
084   * <p>This field is not safe for concurrent use by multiple
085   * threads.</p>
086   *
087   * @see #startHttpServers(Object, BeanManager)
088   */
089  private final Collection<HttpServer> startedHttpServers;
090
091
092  /*
093   * Constructors.
094   */
095
096
097  /**
098   * Creates a new {@link HttpServerStartingExtension}.
099   *
100   * @see AbstractBlockingExtension#AbstractBlockingExtension()
101   */
102  public HttpServerStartingExtension() {
103    super();
104    this.startedHttpServers = new LinkedList<>();
105  }
106
107
108  /*
109   * Instance methods.
110   */
111
112
113  /**
114   * Upon startup ({@linkplain Initialized initialization} of the
115   * {@linkplain ApplicationScoped application scope}), looks for
116   * {@link HttpServer} instances in the {@linkplain BeanManager CDI
117   * container} represented by the supplied {@link BeanManager} and
118   * {@linkplain HttpServer#start() starts them}.
119   *
120   * <p>If any {@link HttpServer}s were {@linkplain HttpServer#start()
121   * started}, then a {@linkplain #fireBlockingEvent(BeanManager)
122   * blocking event is fired}.</p>
123   *
124   * @param event the event signaling that the CDI container has
125   * started; ignored; may be {@code null}
126   *
127   * @param beanManager a {@link BeanManager} supplied by the current
128   * CDI container; may be {@code null} in which case no action will
129   * be taken
130   *
131   * @exception IOException if an error occurs while {@linkplain
132   * HttpServer#start() starting the <code>HttpServer</code>
133   * instances}
134   *
135   * @see HttpServer#start()
136   *
137   * @see #fireBlockingEvent(BeanManager)
138   */
139  private final void startHttpServers(@Observes @Initialized(ApplicationScoped.class) @Priority(LIBRARY_AFTER) final Object event, final BeanManager beanManager) throws IOException {
140    if (this.logger.isTraceEnabled()) {
141      this.logger.trace("ENTRY {} {} {}, {}", this.getClass().getName(), "startHttpServers", event, beanManager);
142    }
143    if (beanManager != null) {
144      final Instance<Object> beans = beanManager.createInstance();
145      assert beans != null;
146      final Instance<HttpServer> httpServers = beans.select(HttpServer.class);
147      assert httpServers != null;
148      if (!httpServers.isUnsatisfied()) {
149        synchronized (this.startedHttpServers) {
150          try {
151            for (final HttpServer httpServer : httpServers) {
152              if (httpServer != null) {
153
154                // This (asynchronous) method starts a daemon thread
155                // in the background; see
156                // https://github.com/GrizzlyNIO/grizzly-mirror/blob/2.3.x/modules/http-server/src/main/java/org/glassfish/grizzly/http/server/HttpServer.java#L816
157                // and work backwards to the start() method.  Among
158                // other things, that won't prevent the JVM from
159                // exiting normally.  Think about that for a while.
160                // We'll need another mechanism to block the CDI
161                // container from simply shutting down.
162                httpServer.start();
163                
164                // We store our own list of started HttpServers to use
165                // in other methods in this class rather than relying
166                // on Instance<HttpServer> because we don't know what
167                // scope the HttpServer instances in question are.
168                // For example, they might be in Dependent scope,
169                // which would mean every Instance#get() invocation
170                // might create a new one.
171                this.startedHttpServers.add(httpServer);
172                
173              }
174            }
175          } catch (final IOException throwMe) {
176            if (!this.startedHttpServers.isEmpty()) {
177              final Iterator<HttpServer> iterator = this.startedHttpServers.iterator();
178              assert iterator != null;
179              while (iterator.hasNext()) {
180                final HttpServer httpServer = iterator.next();
181                if (httpServer != null && httpServer.isStarted()) {
182                  try {
183                    httpServer.shutdownNow();
184                    iterator.remove();
185                  } catch (final RuntimeException problem) {
186                    throwMe.addSuppressed(problem);
187                  }                  
188                }
189              }
190            }
191            throw throwMe;
192          }
193          if (!this.startedHttpServers.isEmpty()) {
194
195            // Here we fire an *asynchronous* event that will cause
196            // the thread it is received on to block (see the block()
197            // method below).  This is key: the JVM will be prevented
198            // from exiting, and the thread that the event is received
199            // on is (a) a non-daemon thread and (b) managed by the
200            // container.  This is, in other words, a clever way to
201            // take advantage of the CDI container's mandated thread
202            // management behavior so that the CDI container stays up
203            // for at least as long as the HttpServer we spawned
204            // above.
205            this.fireBlockingEvent(beanManager);
206            
207          }
208        }
209      }
210    }
211    if (logger != null && logger.isTraceEnabled()) {
212      logger.trace("EXIT {} {}", this.getClass().getName(), "startHttpServers");
213    }
214  }
215
216  /**
217   * Calls {@link HttpServer#shutdownNow()} on any {@link HttpServer}
218   * instances that were {@linkplain #startHttpServers(Object,
219   * BeanManager) started}.
220   *
221   * @param event the event signaling that the current CDI container
222   * is about to shut down; may be {@code null}; ignored
223   *
224   * @see #startHttpServers(Object, BeanManager)
225   *
226   * @see HttpServer#shutdownNow()
227   */
228  private final void stopHttpServers(@Observes final BeforeShutdown event) {
229    synchronized (this.startedHttpServers) {
230      final Iterator<HttpServer> iterator = this.startedHttpServers.iterator();
231      assert iterator != null;
232      while (iterator.hasNext()) {
233        final HttpServer httpServer = iterator.next();
234        iterator.remove();
235        if (httpServer != null && httpServer.isStarted()) {          
236          httpServer.shutdownNow();
237        }
238      }
239    }
240  }
